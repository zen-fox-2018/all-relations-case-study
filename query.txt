1. CREATE TABLE
CREATE TABLE "Manager" ( `id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `manager_name` TEXT )
CREATE TABLE `Karyawan` ( `id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `NIK` INTEGER, `nama_depan` TEXT, `nama_belakang` TEXT, `jenis_kelamin` TEXT, `email` TEXT, `phone_number` TEXT )
CREATE TABLE `Project` ( `id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `project_name` TEXT )
create table Departemen (
	department_name TEXT(50),
	manager_id INT,
	employee_id INT
);
create table ProjectEmployee (
	project_id INT,
	employee_id INT
);

5. QUERY UNTUK MENGETAHUI MANAGER MEMBAWAHU DEPARTEMENT APA SAJA

6. QUERY UNTUK MENGETAHUI 1 KARYAWAN MENGERJAKAN PROJECT APA SAJA
SELECT Karyawan.nama_depan, Project.project_name FROM Karyawan JOIN ProjectEmployee ON Karyawan.id = ProjectEmployee.project_id JOIN Project ON ProjectEmployee.project_id = Project.id

7. QUERY UNTUK MENGETAHUI 1 PROJECT DIKERJAKAN OLEH KARYAWAN YANG MANA SAJA
SELECT Project.project_name, Karyawan.nama_depan FROM ProjectEmployee JOIN Karyawan ON ProjectEmployee.employee_id = Karyawan.id 
JOIN Project ON ProjectEmployee.project_id = Project.id
